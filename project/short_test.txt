# это тестовая геометрическая ggb-конструкция на кратком языке short (очень сокращенный python)

x = 5                   # переменная (объект Var)
y = x + 2               # переменная
Smth = Point(x, y)      # точка с заданными координатами и использованием переменных

A = Point(1.5, 2)       # точка с заданными координатами в виде чисел
B = Point(XCoord(Smth), YCoord(Smth) - 2)   # точка с заданными координатами со вложенными командами

ω = Circle(A, B)        # окружность по центру A и точке B на окружности
r = Radius(ω)           # переменной r присваевается значение команды Radius
C = Point(ω)            # случайная точка на окружности

D = Point(Line(A, B))   
E = Intersect(ω, Line(C, D))
segm1 = Segment(D, E)

f(x) = x^2              # заданная функция (объект Expression)
poly1 = Polygon(segm1, f(2))    # правильный многоугольник со стороной segm1 и количеством вершин f(2) = 4

[E, E1] = Intersect(ω, Line(C, D))
