# это тестовая геометрическая ggb-конструкция на кратком языке shortpy (очень сокращенный python)

x = 5                   # присвоение переменной x значения 5
y = x + 2               # присвоение переменной y значения выражения "x + 5"
Smth = Point(x, y)      # точка с заданными координатами с использованием переменных

A = Point(1.5, 2)       # точка с заданными координатами в виде чисел
B = Point(XCoord(Smth), YCoord(Smth) - 2)   # точка с заданными координатами со вложенными командами и выражением

ω = Circle(A, B)        # окружность по центру A и точке B на окружности
r = Radius(ω)           # присвоение переменной r значения команды Radius
C = Point(ω)            # случайная точка C на окружности
Point(ω)                # еще одна случайная точка на окружности, с системным именем

D = Point(Line(A, B))   # точка на прямой с использованием вложенной команды
E = Intersect(ω, Line(C, D))    # первая точка пересечения окружности и прямой с использованием вложенной команды
segm1 = Segment(D, E)   # отрезок по двум точкам

f(x) = x^2              # присвоение переменной f выражения с переменной x
poly1 = Polygon(segm1, f(2))    # правильный многоугольник со стороной segm1 и количеством вершин f(2) = 4

[E, E1] = Intersect(ω, Line(C, D))  # две точки пересечения окружности и прямой
[poly1, A1, A2, A3, A4] = Polygon(segm1, 4)     # правильный многоугольник по отрезку и количеству вершин
